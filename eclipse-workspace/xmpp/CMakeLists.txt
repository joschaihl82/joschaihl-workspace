cmake_minimum_required(VERSION 3.16)
project(xmpp_voip_expanded LANGUAGES CXX)

# Use C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Enable automoc so Qt's moc runs automatically for QObject classes
set(CMAKE_AUTOMOC ON)
# Ensure the current source dir and build dir are in include path for generated moc files
set(CMAKE_INCLUDE_CURRENT_DIR ON)

# Optionally set build type
if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
endif()

# Find Qt6 components
find_package(Qt6 COMPONENTS Widgets Network Multimedia REQUIRED)

# Find pkg-config and QXmpp via pkg-config (adjust if your system uses a different mechanism)
find_package(PkgConfig REQUIRED)
pkg_check_modules(QXMPP REQUIRED IMPORTED_TARGET qxmpp)

# Source file(s)
set(SRC
  xmpp.cpp
)

# Create executable
add_executable(${PROJECT_NAME} ${SRC})

# Make sure the build directory is on the include path so "xmpp.moc" (generated by automoc)
# can be found by the compiler when the source contains #include "xmpp.moc"
target_include_directories(${PROJECT_NAME}
  PRIVATE
    ${CMAKE_CURRENT_BINARY_DIR}
)

# Link Qt and QXmpp and ALSA
target_link_libraries(${PROJECT_NAME}
  PRIVATE
    Qt6::Widgets
    Qt6::Network
    Qt6::Multimedia
    PkgConfig::QXMPP
    asound
)

# Compiler warnings and useful flags
if(MSVC)
  target_compile_options(${PROJECT_NAME} PRIVATE /W4 /permissive-)
else()
  target_compile_options(${PROJECT_NAME} PRIVATE -Wall -Wextra -Wpedantic -Wno-unused-parameter)
endif()

# RPATH for development convenience
set_target_properties(${PROJECT_NAME} PROPERTIES
  BUILD_RPATH "$ORIGIN/../lib"
  INSTALL_RPATH "$ORIGIN/../lib"
)

# Install rule
install(TARGETS ${PROJECT_NAME} RUNTIME DESTINATION bin)

# Helpful messages
message(STATUS "Project: ${PROJECT_NAME}")
message(STATUS "C++ standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "Qt6 Widgets: ${Qt6Widgets_FOUND}")
message(STATUS "QXmpp pkg-config: ${QXMPP_FOUND}")
